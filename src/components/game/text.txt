// src/components/game/PuzzleGrid.jsx
import { useState } from "react";
import GridCell from "./GridCell";
import PuzzleTile from "./PuzzleTile";

export default function PuzzleGrid({ imageSrc, gridSize = 3 }) {
  const totalTiles = gridSize * gridSize;

function shuffle(array) {
  const copy = [...array];
  for (let i = copy.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [copy[i], copy[j]] = [copy[j], copy[i]];
  }
  return copy;
}


  const [availableTiles, setAvailableTiles] = useState(
    shuffle(Array.from({ length: totalTiles }, (_, i) => i))
  );
  const [placedTiles, setPlacedTiles] = useState(Array(totalTiles).fill(null));

  function handleDrop(tileIndex, cellIndex) {
    if (placedTiles[cellIndex] === null) {
      const newPlaced = [...placedTiles];
      newPlaced[cellIndex] = tileIndex;
      setPlacedTiles(newPlaced);
      setAvailableTiles((prev) => prev.filter((t) => t !== tileIndex));
    }
  }

  const isSolved = placedTiles.every((val, i) => val === i);

  return (
    <div className="relative flex flex-row ">
      <div
        className="grid w-[300px] h-[300px]  rounded"
        style={{ gridTemplateColumns: `repeat(${gridSize}, 1fr)` }}
      >
        {placedTiles.map((tile, i) => (
          <GridCell
            key={i}
            cellIndex={i}
            placedTile={tile}
            imageSrc={imageSrc}
            gridSize={gridSize}
            onDrop={handleDrop}
          />
        ))}
      </div>
      <div className="grid grid-cols-3">
      {(availableTiles.map((tileIndex) => (
        <PuzzleTile
          key={tileIndex}
          index={tileIndex}
          imageSrc={imageSrc}
          gridSize={gridSize}
        />
      )))}
      </div>
      {isSolved && (
        <div className="absolute top-0 left-0 w-full h-full bg-green-500 bg-opacity-70 flex items-center justify-center text-white text-2xl font-bold">
          🎉 Puzzle Solved!
        </div>
      )}
    </div>
  );
}
